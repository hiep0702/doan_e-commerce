//CREATE DATABASE

CREATE DATABASE Eproject;
USE Eproject;

CREATE TABLE Admin(
	Admin_id int auto_increment primary key
	,Admin_firstName varchar(10)
	,Admin_lastName varchar(10)
	,Admin_phoneNumber int(11)
	,Admin_email varchar(100)
	,Admin_userName varchar(10)
	,Admin_password_hash varchar(20)
);

CREATE TABLE Customers(
	Customer_ID int auto_increment primary key
    ,Customer_first_name varchar(20)
    ,Customer_last_name varchar(20)
	,Customer_DOB datetime
    ,Customer_phone varchar(20)
	,Customer_email varchar(200)
	,User_name  varchar(20)
	,Password_hash varchar(200)
	,Customer_rank varchar(10)
	,Create_at  datetime      
	,Update_at  datetime  
);

CREATE TABLE Brands(
	Brand_ID int auto_increment primary key
	,Brand_name varchar(50)
	,Brand_logo varchar(500)
	,Brand_slide_img varchar(500)
	,Brand_video_demo varchar(500)
	,Brand_information text
	,Brand_history text
	,Brand_slug varchar(20)
	,Create_at datetime      
	,Update_at datetime    
);

CREATE TABLE Categories(
	Category_ID int auto_increment primary key 
    ,Category_name varchar(50)
    ,Category_logo varchar(500)
    ,Category_slide_img varchar(500)
    ,Category_video_demo varchar(500)
    ,Category_informations varchar(200)
    ,Category_history varchar(200)
    ,Create_at datetime
    ,Update_at datetime
    ,Brand_ID int references Brand(Brand_ID)
);


CREATE TABLE Products(
	Product_ID int auto_increment primary key
    ,Product_name varchar(200)
    ,Category_ID int references Categories(Category_ID)
	,Brand_ID int references Brand(Brand_ID)
);

CREATE TABLE Product_details(
	Products_details_ID int auto_increment primary key
	,Product_import_price float
	,Product_export_price float
	,Product_sales_price float
	,Product_main_img varchar(500)
	,Product_slide_img varchar(500)
	,Product_video_demo varchar(500)
	,Product_informations text
	,Create_at datetime      
	,Update_at datetime     
	,in_stock int
	,product_materials text
	,Product_history varchar(200)
	,Product_colors varchar(100)
	,Products_ID int references Products(Product_ID)
);

CREATE TABLE Wish_list(
	List_ID  int auto_increment primary key
	,quantity int
	,Product_ID int references Porducts(Product_ID)
	,Customer_ID int references Customers(Customer_ID)
);

CREATE TABLE Orders(
	Order_ID int auto_increment primary key
    ,Order_date datetime
	,Order_status varchar(20)
	,Order_payment_method varchar(20)
	,Order_receving_method varchar(20)
	,Location varchar(200)
	,Customer_ID int references Customers(Customer_ID)
);

CREATE TABLE Oders_details(
	Order_Details_ID int auto_increment primary key
	,Product_quantity int
	,Order_ID int references Orders(Order_ID)
    ,Product_ID  int references Products(Product_ID)
    ,Product_export_price float references product_details(Product_export_price)
);

CREATE TABLE Product_review(
	Review_ID int auto_increment primary key
	,Review_content text
	,Product_ID  int references Products(Product_ID)
	,Customer_ID int references Customers(Customer_ID)
    ,Order_ID int references Orders(Order_ID)
);
